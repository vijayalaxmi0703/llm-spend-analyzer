# Credex Audit — runtime requirements
# This project is a Node.js / Next.js application. Keep these engine and dependency hints

# Node & npm
node >= 20.0.0
npm >= 10.0.0

# Core dependencies (exact versions in package.json)
next ^15.5.18
react 18.3.1
react-dom 18.3.1
typescript ^5.5.4
tailwindcss ^3.4.5
@supabase/supabase-js ^2.40.0
@anthropic-ai/sdk ^0.98.0 (optional)
zod ^3.23.0
react-hook-form ^7.58.0

# Dev / QA tools
vitest ^4.1.7
eslint + eslint-config-next
lighthouse (manual audits)

# External services (configure via .env.local)
- Supabase (audits + leads tables) — required for cross-device sharing
- Resend (transactional email) — optional
- Anthropic API — optional (executive summary)

# To run locally
1. Copy env: `cp .env.example .env.local`
2. Install: `npm install`
3. Start: `npm run dev`
